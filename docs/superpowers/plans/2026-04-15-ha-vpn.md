```python
@dataclass
class Peer:
    name: str
    public_key: str
    preshared_key: str
    allowed_ips: str


def reverse_key(key: str) -> str:
    """Reverse a WireGuard base64 key at the byte level. Self-inverse."""
    try:
        raw = base64.b64decode(key)
        return base64.b64encode(raw[::-1]).decode()
    except Exception:
        stripped = key.rstrip("=")
        padding = key[len(stripped):]
        return stripped[::-1] + padding


def reverse_peer_keys(content: str, name: str) -> str:
    """
    Reverse PublicKey and PresharedKey in the [Peer] block for `name`.
    Uses split('\\n') (not splitlines) to preserve trailing newlines.
    Self-inverse: calling twice restores the original.
    """
    lines = content.split("\n")
    result = []
    in_target = False

    for line in lines:
        stripped = line.strip()

        if stripped.startswith("# Client") and "=" in stripped:
            peer_name = stripped.split("=", 1)[1].strip()
            in_target = (peer_name == name)
        elif stripped == "[Peer]" and not in_target:
            in_target = False

        if in_target and not stripped.startswith("#"):
            lower = stripped.lower()
            if lower.startswith("publickey") or lower.startswith("presharedkey"):
                key_part, sep, value = line.partition("=")
                if sep and value.strip():
                    result.append(f"{key_part}= {reverse_key(value.strip())}")
                    continue

        result.append(line)

    return "\n".join(result)


def parse_peers(content: str) -> list[Peer]:
    """
    Parse all [Peer] blocks from a WireGuard server config.
    Each block is preceded by a '# Client = <name>' comment.
    """
    peers = []
    lines = content.split("\n")
    current_name: Optional[str] = None
    current: dict = {}

    for line in lines:
        stripped = line.strip()

        if stripped.startswith("# Client") and "=" in stripped:
            current_name = stripped.split("=", 1)[1].strip()
            current = {}
        elif stripped == "[Peer]":
            pass  # name was already captured from preceding comment
        elif current_name is not None:
            if stripped.lower().startswith("publickey"):
                current["public_key"] = stripped.split("=", 1)[1].strip()
            elif stripped.lower().startswith("presharedkey"):
                current["preshared_key"] = stripped.split("=", 1)[1].strip()
            elif stripped.lower().startswith("allowedips"):
                current["allowed_ips"] = stripped.split("=", 1)[1].strip()
                if "public_key" in current and "preshared_key" in current:
                    peers.append(Peer(
                        name=current_name,
                        public_key=current["public_key"],
                        preshared_key=current["preshared_key"],
                        allowed_ips=current["allowed_ips"],
                    ))
                current_name = None
                current = {}

    return peers


def next_free_ip(peers: list[Peer], subnet: str) -> str:
    """
    Find first available client IP in subnet.
    Server is at .1; clients start at .2.
    """
    network = ipaddress.IPv4Network(subnet, strict=False)
    used = set()
    for p in peers:
        ip_str = p.allowed_ips.split("/")[0].strip()
        try:
            used.add(ipaddress.IPv4Address(ip_str))
        except ValueError:
            pass

    for host in network.hosts():
        if str(host).endswith(".1"):
            continue
        if host not in used:
            return str(host)

    raise ValueError(f"No free IPs in {subnet}")


def render_server_conf(iface: str, peers: list[Peer], server_privkey: str, address: str) -> str:
    """Render a complete WireGuard server config from peer list."""
    wg_port, _ = _PORTS[iface]
    lines = [
        "[Interface]",
        f"PrivateKey = {server_privkey}",
        f"Address = {address}",
        f"ListenPort = {wg_port}",
        "",
    ]
    for peer in peers:
        lines += [
            f"# Client = {peer.name}",
            "[Peer]",
            f"PublicKey = {peer.public_key}",
            f"PresharedKey = {peer.preshared_key}",
            f"AllowedIPs = {peer.allowed_ips}",
            "",
        ]
    return "\n".join(lines)


def render_client_conf(
    peer: Peer,
    iface: str,
    server_pubkey: str,
    endpoint_host: str,
    flavor: str,
) -> str:
    """
    Render a client .conf file.
    flavor='awg' → include AWG obfuscation fields (Jc/Jmin/Jmax/S1/S2/H1-H4/I1)
    flavor='wg'  → standard WireGuard only
    """
    wg_port, awg_port = _PORTS[iface]
    client_ip = peer.allowed_ips.split("/")[0]
    port = awg_port if flavor == "awg" else wg_port

    # Derive client private key from public key reversal (for display only;
    # actual private key is stored in VPNConfig.config_metadata)
    # In practice, caller passes the real private key via peer.name lookup.
    # This function renders with what it's given — caller must pass private key.
    # We use a placeholder here; actual callers pass real data.
    lines = ["[Interface]"]

    # AWG obfuscation fields come right after PrivateKey in AWG flavor
    if flavor == "awg":
        lines += [
            "PrivateKey = ${CLIENT_PRIVATE_KEY}",
            f"Address = {client_ip}/32",
            "DNS = 10.29.8.1",
            "Jc = 100",
            "Jmin = 20",
            "Jmax = 100",
            "S1 = 0",
            "S2 = 0",
            "H1 = 1",
            "H2 = 2",
            "H3 = 3",
            "H4 = 4",
            f"I1 = {_AWG_I1}",
        ]
    else:
        lines += [
            "PrivateKey = ${CLIENT_PRIVATE_KEY}",
            f"Address = {client_ip}/32",
            "DNS = 10.29.8.1",
        ]

    lines += [
        "",
        "[Peer]",
        f"PublicKey = {server_pubkey}",
        f"PresharedKey = {peer.preshared_key}",
        f"Endpoint = {endpoint_host}:{port}",
    ]

    if iface == "antizapret":
        lines.append("AllowedIPs = 10.29.8.0/24${IPS}")
    else:
        lines.append("AllowedIPs = 0.0.0.0/0, ::/0")

    lines.append("PersistentKeepalive = 15")
    return "\n".join(lines) + "\n"
```

```python
"""
WgBlobStore: thin repository over the wg_file_state table.
All writes trigger pg_notify('wg_file_state_changed', path) via DB trigger.
"""
import hashlib
from datetime import datetime
from typing import Optional
from sqlalchemy.orm import Session
from sqlalchemy.dialects.postgresql import insert as pg_insert
from sqlalchemy import text

from app.db.models import WgFileState


class WgBlobStore:
    def __init__(self, db: Session):
        self._db = db

    def get(self, path: str) -> Optional[bytes]:
        row = self._db.get(WgFileState, path)
        return row.content if row else None

    def put(self, path: str, content: bytes, by: str) -> None:
        sha = hashlib.sha256(content).hexdigest()
        now = datetime.utcnow()

        existing = self._db.get(WgFileState, path)
        if existing:
            existing.content = content
            existing.sha256 = sha
            existing.size_bytes = len(content)
            existing.updated_at = now
            existing.updated_by = by
        else:
            row = WgFileState(
                path=path,
                content=content,
                sha256=sha,
                size_bytes=len(content),
                updated_at=now,
                updated_by=by,
            )
            self._db.add(row)

        self._db.commit()

    def get_all_paths(self) -> dict[str, str]:
        """Return {path: sha256} for all stored files."""
        rows = self._db.query(WgFileState).all()
        return {row.path: row.sha256 for row in rows}
```

```python
"""
New DB-backed VPN manager — replaces file-I/O + subprocess approach.
Uses WgBlobStore for all WireGuard config reads/writes.
"""
import base64
import re
import logging
from typing import Optional

from sqlalchemy.orm import Session
from sqlalchemy import text

from app.db.models import WgServerKeys
from app.services.wg_blob_store import WgBlobStore
from app.services.wg_templates import (
    Peer, parse_peers, render_server_conf, render_client_conf,
    next_free_ip, reverse_peer_keys, reverse_key,
)

logger = logging.getLogger(__name__)

CLIENT_NAME_PATTERN = re.compile(r'^[a-zA-Z0-9_-]+$')

_IFACES = [
    ("antizapret", "/etc/wireguard/antizapret.conf", "10.29.8.1/21"),
    ("vpn",        "/etc/wireguard/vpn.conf",        "10.28.8.1/21"),
]

_ADVISORY_LOCK_ID = 8675309  # arbitrary stable integer for WG operations


class VPNManagerError(Exception):
    pass


def _generate_wg_keypair() -> tuple[str, str]:
    """Return (private_key_b64, public_key_b64) using X25519."""
    from cryptography.hazmat.primitives.asymmetric.x25519 import X25519PrivateKey
    priv = X25519PrivateKey.generate()
    priv_b64 = base64.b64encode(priv.private_bytes_raw()).decode()
    pub_b64  = base64.b64encode(priv.public_key().public_bytes_raw()).decode()
    return priv_b64, pub_b64


class VpnManager:

    def _validate_name(self, name: str) -> None:
        if not CLIENT_NAME_PATTERN.match(name):
            raise VPNManagerError(f"Invalid client name: {name!r}")
        if len(name) > 32:
            raise VPNManagerError("Client name too long (max 32 chars)")

    def bootstrap(self, db: Session) -> None:
        """Idempotent: create server keypairs and empty conf blobs if missing."""
        store = WgBlobStore(db)

        for iface, conf_path, address in _IFACES:
            # Keypair
            existing = db.get(WgServerKeys, iface)
            if not existing:
                priv, pub = _generate_wg_keypair()
                db.add(WgServerKeys(iface=iface, private_key=priv, public_key=pub))
                db.commit()

            # Empty conf blob
            if store.get(conf_path) is None:
                keys = db.get(WgServerKeys, iface)
                empty_conf = render_server_conf(iface, [], keys.private_key, address)
                store.put(conf_path, empty_conf.encode(), by="bootstrap")

    def _get_server_keys(self, db: Session, iface: str) -> WgServerKeys:
        keys = db.get(WgServerKeys, iface)
        if not keys:
            raise VPNManagerError(f"Server keys not found for {iface}. Run bootstrap first.")
        return keys

    def add_peer(self, db: Session, name: str) -> dict:
        """
        Add a new peer to both antizapret and vpn confs.
        Uses advisory lock to prevent concurrent IP allocation races.
        Returns dict with client_name, vpn_ip, public_key_antizapret, private_key, preshared_key.
        """
        self._validate_name(name)

        # Generate client keypair
        client_priv, client_pub = _generate_wg_keypair()

        # Generate preshared key (32 random bytes b64)
        import os
        psk = base64.b64encode(os.urandom(32)).decode()

        store = WgBlobStore(db)

        # Advisory lock for the duration of this transaction
        db.execute(text(f"SELECT pg_try_advisory_lock({_ADVISORY_LOCK_ID})"))

        try:
            # Determine IP from antizapret conf (canonical source)
            az_iface, az_path, az_addr = _IFACES[0]
            az_content = store.get(az_path) or b""
            az_peers = parse_peers(az_content.decode())
            ip = next_free_ip(az_peers, "10.29.8.0/21")

            for iface, conf_path, address in _IFACES:
                keys = self._get_server_keys(db, iface)
                content = store.get(conf_path) or b""
                peers = parse_peers(content.decode())

                new_peer = Peer(
                    name=name,
                    public_key=client_pub,
                    preshared_key=psk,
                    allowed_ips=f"{ip}/32",
                )
                peers.append(new_peer)

                new_conf = render_server_conf(iface, peers, keys.private_key, address)
                store.put(conf_path, new_conf.encode(), by=f"add_peer:{name}")

        finally:
            db.execute(text(f"SELECT pg_advisory_unlock({_ADVISORY_LOCK_ID})"))

        return {
            "client_name": name,
            "vpn_ip": ip,
            "private_key": client_priv,
            "public_key_antizapret": client_pub,
            "preshared_key": psk,
        }

    def delete_peer(self, db: Session, name: str) -> None:
        self._validate_name(name)
        store = WgBlobStore(db)

        for iface, conf_path, address in _IFACES:
            keys = self._get_server_keys(db, iface)
            content = store.get(conf_path)
            if not content:
                continue
            peers = [p for p in parse_peers(content.decode()) if p.name != name]
            new_conf = render_server_conf(iface, peers, keys.private_key, address)
            store.put(conf_path, new_conf.encode(), by=f"delete_peer:{name}")

    def disable_peer(self, db: Session, name: str) -> None:
        self._validate_name(name)
        store = WgBlobStore(db)

        for iface, conf_path, _addr in _IFACES:
            content = store.get(conf_path)
            if not content:
                continue
            text_conf = content.decode()
            if f"# Client = {name}" not in text_conf:
                continue
            new_conf = reverse_peer_keys(text_conf, name)
            store.put(conf_path, new_conf.encode(), by=f"disable_peer:{name}")

    def enable_peer(self, db: Session, name: str) -> None:
        self._validate_name(name)
        store = WgBlobStore(db)

        for iface, conf_path, _addr in _IFACES:
            content = store.get(conf_path)
            if not content:
                continue
            text_conf = content.decode()
            if f"# Client = {name}" not in text_conf:
                continue
            new_conf = reverse_peer_keys(text_conf, name)
            store.put(conf_path, new_conf.encode(), by=f"enable_peer:{name}")

    def list_peers(self, db: Session) -> list[dict]:
        store = WgBlobStore(db)
        _, az_path, _ = _IFACES[0]
        content = store.get(az_path)
        if not content:
            return []
        peers = parse_peers(content.decode())
        return [
            {"name": p.name, "public_key": p.public_key, "allowed_ips": p.allowed_ips}
            for p in peers
        ]

    def get_client_conf(
        self, db: Session, name: str, flavor: str, endpoint_host: str, iface: str = "antizapret"
    ) -> str:
        store = WgBlobStore(db)
        _, conf_path, _ = next(t for t in _IFACES if t[0] == iface)
        content = store.get(conf_path)
        if not content:
            raise VPNManagerError("Config not initialized")
        peers = parse_peers(content.decode())
        peer = next((p for p in peers if p.name == name), None)
        if not peer:
            raise VPNManagerError(f"Peer {name!r} not found")
        keys = self._get_server_keys(db, iface)
        return render_client_conf(peer, iface, keys.public_key, endpoint_host, flavor)


def generate_client_name(username: str, existing_names: list[str]) -> str:
    base = re.sub(r'[^a-zA-Z0-9_-]', '_', username.split("@")[0])
    max_n = 0
    pattern = re.compile(rf'^{re.escape(base)}-(\d+)$')
    for n in existing_names:
        m = pattern.match(n)
        if m:
            max_n = max(max_n, int(m.group(1)))
    return f"{base}-{max_n + 1}"
```

```python
        # In test environments (SQLite) skip asyncpg
        if not database_url.startswith("postgresql"):
            yield ": keepalive\n\n"
            return

            import asyncpg
            queue: asyncio.Queue = asyncio.Queue()

            conn = await asyncpg.connect(database_url)

            async def on_notify(conn, pid, channel, payload):
                await queue.put(payload)

            await conn.add_listener("wg_file_state_changed", on_notify)

            try:
                while True:
                    try:
                        path = await asyncio.wait_for(queue.get(), timeout=15.0)
                        yield f"data: {{\"path\": \"{path}\"}}\n\n"
                    except asyncio.TimeoutError:
                        yield ": keepalive\n\n"
            finally:
                await conn.remove_listener("wg_file_state_changed", on_notify)
                await conn.close()
        except Exception as e:
            logger.error(f"SSE error: {e}")
            yield f"data: {{\"error\": \"stream closed\"}}\n\n"

        stream(),
```

```python
#!/usr/bin/env python3
"""
CorpWeb Sync Agent — runs on each data-plane node.
Syncs WireGuard configs from control-plane via SSE + heartbeat.
Dependencies: stdlib + requests
"""
import base64
import hashlib
import json
import logging
import os
import subprocess
import sys
import tempfile
import threading
import time
from datetime import datetime
from pathlib import Path

import requests

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
log = logging.getLogger("sync-agent")

ENV_FILE = "/etc/corpweb-sync-agent.env"

# Files managed by this agent and their apply hooks
MANAGED_FILES: dict[str, str | None] = {
    "/etc/wireguard/antizapret.conf": "wg",
    "/etc/wireguard/vpn.conf":        "wg",
    "/root/antizapret/setup":          None,
    "/root/antizapret/config/include-hosts.txt":         "doall",
    "/root/antizapret/config/exclude-hosts.txt":         "doall",
    "/root/antizapret/config/include-ips.txt":           "doall",
    "/root/antizapret/config/exclude-ips.txt":           "doall",
    "/root/antizapret/config/allow-ips.txt":             "doall",
    "/root/antizapret/config/forward-ips.txt":           "doall",
    "/root/antizapret/config/include-adblock-hosts.txt": "doall",
    "/root/antizapret/config/exclude-adblock-hosts.txt": "doall",
    "/root/antizapret/config/remove-hosts.txt":          "doall",
}

_doall_timer: threading.Timer | None = None
_doall_lock = threading.Lock()
DOALL_DEBOUNCE = 5.0  # seconds


def load_env() -> dict:
    cfg = {}
    with open(ENV_FILE) as f:
        for line in f:
            line = line.strip()
            if line and "=" in line and not line.startswith("#"):
                k, _, v = line.partition("=")
                cfg[k.strip()] = v.strip()
    required = ["CONTROL_PLANE_URL", "AGENT_TOKEN", "AGENT_HOSTNAME"]
    for k in required:
        if k not in cfg:
            raise RuntimeError(f"Missing {k} in {ENV_FILE}")
    return cfg


def sha256(data: bytes) -> str:
    return hashlib.sha256(data).hexdigest()


def write_atomic(path: str, content: bytes) -> None:
    p = Path(path)
    p.parent.mkdir(parents=True, exist_ok=True)
    fd, tmp = tempfile.mkstemp(dir=p.parent)
    try:
        os.write(fd, content)
        os.close(fd)
        os.replace(tmp, path)
    except Exception:
        os.unlink(tmp)
        raise


def run_wg_syncconf(path: str) -> None:
    iface = Path(path).stem  # antizapret or vpn
    try:
        subprocess.run(
            ["bash", "-c", f"wg syncconf {iface} <(wg-quick strip {iface})"],
            check=True, timeout=15,
        )
        log.info(f"wg syncconf {iface} OK")
    except Exception as e:
        log.warning(f"wg syncconf {iface} failed: {e}")


def schedule_doall() -> None:
    global _doall_timer
    with _doall_lock:
        if _doall_timer:
            _doall_timer.cancel()
        _doall_timer = threading.Timer(DOALL_DEBOUNCE, _run_doall)
        _doall_timer.daemon = True
        _doall_timer.start()


def _run_doall() -> None:
    log.info("Running doall.sh")
    try:
        subprocess.run(
            ["/root/antizapret/doall.sh"],
            check=True, timeout=300,
        )
        log.info("doall.sh completed")
    except Exception as e:
        log.error(f"doall.sh failed: {e}")


def apply_path(path: str, content: bytes, local_sha: str) -> None:
    remote_sha = sha256(content)
    if remote_sha == local_sha:
        log.debug(f"Skip {path} (sha matches)")
        return

    write_atomic(path, content)
    log.info(f"Applied {path}")

    hook = MANAGED_FILES.get(path)
    if hook == "wg":
        run_wg_syncconf(path)
    elif hook == "doall":
        schedule_doall()


class Agent:
    def __init__(self, cfg: dict):
        self.base = cfg["CONTROL_PLANE_URL"].rstrip("/")
        self.token = cfg["AGENT_TOKEN"]
        self.hostname = cfg["AGENT_HOSTNAME"]
        self.session = requests.Session()
        self.session.headers["Authorization"] = f"Bearer {self.token}"

    def register(self) -> None:
        resp = self.session.post(
            f"{self.base}/api/v1/agent/register",
            json={"hostname": self.hostname, "private_ip": self._local_ip()},
            timeout=30,
        )
        resp.raise_for_status()
        data = resp.json()
        log.info(f"Registered as node_id={data['node_id']}")

        # Write server keys if provided
        keys = data.get("wg_server_keys", {})
        for iface, k in keys.items():
            keydir = Path("/etc/wireguard")
            keyfile = keydir / f"{iface}.key"
            pubfile = keydir / f"{iface}.key.pub"
            # Only write if different (avoid unnecessary wg restart)
            if not keyfile.exists() or keyfile.read_text().strip() != k["private_key"]:
                write_atomic(str(keyfile), k["private_key"].encode())
                write_atomic(str(pubfile), k["public_key"].encode())
                os.chmod(str(keyfile), 0o600)
                log.info(f"Wrote keypair for {iface}")

    def _local_ip(self) -> str:
        import socket
        try:
            s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
            s.connect(("8.8.8.8", 80))
            return s.getsockname()[0]
        except Exception:
            return "127.0.0.1"

    def startup_reconcile(self) -> None:
        log.info("Starting reconcile of all managed files")
        for path in MANAGED_FILES:
            try:
                resp = self.session.get(
                    f"{self.base}/api/v1/agent/file",
                    params={"path": path},
                    timeout=30,
                )
                if resp.status_code == 404:
                    log.debug(f"File not in CP yet: {path}")
                    continue
                resp.raise_for_status()
                data = resp.json()
                content = base64.b64decode(data["content"])

                local_sha = ""
                p = Path(path)
                if p.exists():
                    local_sha = sha256(p.read_bytes())

                apply_path(path, content, local_sha)
            except Exception as e:
                log.warning(f"Reconcile failed for {path}: {e}")

    def send_heartbeat(self) -> None:
        try:
            self.session.post(
                f"{self.base}/api/v1/agent/heartbeat",
                json={
                    "applied_sha": self._local_shas(),
                    "health": "ok",
                    "metrics": self._collect_metrics(),
                },
                timeout=10,
            )
        except Exception as e:
            log.debug(f"Heartbeat failed: {e}")

    def _local_shas(self) -> dict:
        out = {}
        for path in MANAGED_FILES:
            p = Path(path)
            if p.exists():
                out[path] = sha256(p.read_bytes())
        return out

    def _collect_metrics(self) -> dict:
        metrics: dict = {}
        for iface in ("antizapret", "vpn"):
            try:
                result = subprocess.run(
                    ["wg", "show", iface, "latest-handshakes"],
                    capture_output=True, text=True, timeout=5,
                )
                now = time.time()
                active = sum(
                    1 for line in result.stdout.splitlines()
                    if line.strip() and (now - int(line.split()[1])) < 180
                )
                metrics[f"active_peers_{iface}"] = active
            except Exception:
                pass

        # rx/tx from /proc/net/dev
        try:
            with open("/proc/net/dev") as f:
                for line in f:
                    if "eth0" in line or "ens" in line:
                        parts = line.split()
                        metrics["rx_bytes"] = int(parts[1])
                        metrics["tx_bytes"] = int(parts[9])
                        break
        except Exception:
            pass

        return metrics

    def stream_events(self) -> None:
        log.info("Connecting to SSE stream")
        with self.session.get(
            f"{self.base}/api/v1/agent/events",
            stream=True,
            timeout=None,
        ) as resp:
            resp.raise_for_status()
            last_heartbeat = time.time()

            for line in resp.iter_lines(decode_unicode=True):
                if not line:
                    continue
                if line.startswith(":"):
                    # keepalive comment
                    pass
                elif line.startswith("data:"):
                    payload = line[5:].strip()
                    try:
                        data = json.loads(payload)
                        path = data.get("path")
                        if path and path in MANAGED_FILES:
                            file_resp = self.session.get(
                                f"{self.base}/api/v1/agent/file",
                                params={"path": path},
                                timeout=30,
                            )
                            file_resp.raise_for_status()
                            fdata = file_resp.json()
                            content = base64.b64decode(fdata["content"])
                            p = Path(path)
                            local_sha = sha256(p.read_bytes()) if p.exists() else ""
                            apply_path(path, content, local_sha)
                    except Exception as e:
                        log.warning(f"Event handling error: {e}")

                now = time.time()
                if now - last_heartbeat >= 30:
                    self.send_heartbeat()
                    last_heartbeat = now


def main() -> None:
    cfg = load_env()
    agent = Agent(cfg)

    agent.register()

    while True:
        try:
            agent.startup_reconcile()
            agent.stream_events()
        except KeyboardInterrupt:
            log.info("Shutting down")
            sys.exit(0)
        except Exception as e:
            log.error(f"Connection lost: {e}. Reconnecting in 5s")
            time.sleep(5)


if __name__ == "__main__":
    main()
```

```bash
#!/usr/bin/env bash
set -euo pipefail

echo "=== CorpAdmin-AZ Control-Plane Installer ==="
echo ""
echo "Select installation mode:"
echo "  [1] Native (recommended for production)"
echo "  [2] Docker"
read -rp "Choice [1/2]: " MODE

DOMAIN=""
read -rp "Panel domain (e.g. panel.example.com): " DOMAIN

case "$MODE" in
  1) install_native ;;
  2) install_docker ;;
  *) echo "Invalid choice"; exit 1 ;;
esac

function install_native() {
    echo "==> Installing native dependencies"
    apt-get update -qq
    apt-get install -y postgresql nginx python3 python3-pip certbot python3-certbot-nginx

    echo "==> Setting up PostgreSQL"
    DB_PASS=$(openssl rand -hex 16)
    sudo -u postgres psql -c "CREATE USER corpweb WITH PASSWORD '$DB_PASS';" 2>/dev/null || true
    sudo -u postgres psql -c "CREATE DATABASE corpweb OWNER corpweb;" 2>/dev/null || true

    echo "==> Installing Python dependencies"
    pip3 install -r /opt/corpweb/requirements.txt

    echo "==> Configuring environment"
    SECRET_KEY=$(openssl rand -hex 32)
    cat > /etc/corpweb-backend.env <<EOF
DATABASE_URL=postgresql://corpweb:$DB_PASS@localhost/corpweb
SECRET_KEY=$SECRET_KEY
EOF
    chmod 600 /etc/corpweb-backend.env

    echo "==> Running migrations"
    cd /opt/corpweb && DATABASE_URL="postgresql://corpweb:$DB_PASS@localhost/corpweb" alembic upgrade head

    echo "==> Bootstrap VPN manager"
    cd /opt/corpweb && DATABASE_URL="postgresql://corpweb:$DB_PASS@localhost/corpweb" python3 -m app.bootstrap

    echo "==> Installing systemd service"
    cat > /etc/systemd/system/corpweb-backend.service <<EOF
[Unit]
Description=CorpWeb Backend
After=postgresql.service

[Service]
EnvironmentFile=/etc/corpweb-backend.env
WorkingDirectory=/opt/corpweb
ExecStart=/usr/local/bin/uvicorn app.main:app --host 127.0.0.1 --port 8000
Restart=on-failure

[Install]
WantedBy=multi-user.target
EOF

    echo "==> Configuring nginx"
    cp /opt/corpweb/control-plane/nginx.conf /etc/nginx/nginx.conf

    echo "==> Obtaining TLS certificate"
    certbot --nginx -d "$DOMAIN" --non-interactive --agree-tos -m "admin@$DOMAIN"

    systemctl daemon-reload
    systemctl enable --now postgresql nginx corpweb-backend

    echo ""
    echo "=== Installation complete ==="
    echo "Panel: https://$DOMAIN"
}

function install_docker() {
    echo "==> Installing Docker"
    apt-get update -qq
    apt-get install -y docker.io docker-compose-plugin certbot

    echo "==> Obtaining TLS certificate (standalone)"
    certbot certonly --standalone -d "$DOMAIN" --non-interactive --agree-tos -m "admin@$DOMAIN"

    echo "==> Writing .env"
    DB_PASS=$(openssl rand -hex 16)
    SECRET_KEY=$(openssl rand -hex 32)
    cat > /opt/corpweb/.env <<EOF
DB_PASSWORD=$DB_PASS
SECRET_KEY=$SECRET_KEY
DOMAIN=$DOMAIN
EOF

    echo "==> Starting services"
    cd /opt/corpweb && docker compose up -d --build

    echo "==> Running migrations"
    docker compose exec backend alembic upgrade head
    docker compose exec backend python3 -m app.bootstrap

    echo ""
    echo "=== Installation complete (Docker) ==="
}
```

```python
# In existing admin.py or new file
from fastapi import UploadFile, File
from typing import Optional

@router.post("/import-wgfiles")
async def import_wgfiles(
    antizapret_conf: Optional[UploadFile] = File(None),
    vpn_conf: Optional[UploadFile] = File(None),
    wg_key: Optional[UploadFile] = File(None),
    db: Session = Depends(get_db),
    _=Depends(get_current_admin_user),
):
    """
    Import WireGuard configs from existing node into DB.
    Called once during migration from single-node to HA setup.
    """
    store = WgBlobStore(db)
    imported = []

    file_map = {
        "antizapret.conf": (antizapret_conf, "/etc/wireguard/antizapret.conf"),
        "vpn.conf":         (vpn_conf,        "/etc/wireguard/vpn.conf"),
    }

    for label, (upload, path) in file_map.items():
        if upload:
            content = await upload.read()
            store.put(path, content, by="import")
            imported.append(label)

    if wg_key:
        key_content = await wg_key.read()
        # Parse private key → derive public key → store in wg_server_keys
        import base64
        from cryptography.hazmat.primitives.asymmetric.x25519 import X25519PrivateKey
        from app.db.models import WgServerKeys
        priv_b64 = key_content.decode().strip()
        priv_raw = base64.b64decode(priv_b64)
        priv_obj = X25519PrivateKey.from_private_bytes(priv_raw)
        pub_b64 = base64.b64encode(priv_obj.public_key().public_bytes_raw()).decode()

        for iface in ("antizapret", "vpn"):
            existing = db.get(WgServerKeys, iface)
            if existing:
                existing.private_key = priv_b64
                existing.public_key = pub_b64
            else:
                db.add(WgServerKeys(iface=iface, private_key=priv_b64, public_key=pub_b64))
        db.commit()
        imported.append("wg_key")

    return {"imported": imported, "count": len(imported)}
```
