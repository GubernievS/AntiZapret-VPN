# HA Migration Plan — Wire DB-backed VPN manager + cutover

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Switch the running wgfi2 system from file/subprocess-based VPN management to DB-backed, so that a DB backup/restore on a new control-plane server gives a fully working system with multi-node support.

**Architecture:** Alembic adds 3 new tables to existing `corpweb_db`. A migration script reads 12 managed files + server keypair + client private keys from disk into the DB. Existing API endpoints switch from `vpn_manager` (subprocess) to `vpn_manager_new` (DB blob). The sync-agent on wgfi2 keeps local files in sync with DB.

**Tech Stack:** Existing FastAPI + SQLAlchemy + PostgreSQL + alembic. No new deps (asyncpg already added in requirements.txt).

---

## Key Design Decision: Client Config Downloads

**Problem:** Antizapret client configs contain a ~15KB AllowedIPs list (anti-censorship routing table). This list is generated by doall.sh and changes when domains are added/removed. Current system stores client configs as files on disk.

**Solution:**
1. Store **client private keys** in `vpn_configs.config_metadata` (extracted from server conf `# PrivateKey` comments during migration)
2. Store the **AllowedIPs template** as a blob in `wg_file_state` at path `antizapret:allowed_ips` (extracted from any existing client conf during migration; updated when doall.sh runs)
3. `render_client_conf` generates configs on-the-fly from stored data
4. Existing client configs on disk become redundant after migration

```python
"""
One-time migration: files on disk → DB.

Usage on wgfi2:
    cd /opt/corpweb/backend
    python3 -m app.migrate

Idempotent: safe to run multiple times (upserts).
"""
import logging
import os
import re
import sys
from pathlib import Path

from sqlalchemy.orm import Session

from app.db.models import WgServerKeys
from app.services.wg_blob_store import WgBlobStore

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
log = logging.getLogger("migrate")

# Files to migrate into wg_file_state
MANAGED_FILES = [
    "/etc/wireguard/antizapret.conf",
    "/etc/wireguard/vpn.conf",
    "/root/antizapret/setup",
    "/root/antizapret/config/include-hosts.txt",
    "/root/antizapret/config/exclude-hosts.txt",
    "/root/antizapret/config/include-ips.txt",
    "/root/antizapret/config/exclude-ips.txt",
    "/root/antizapret/config/allow-ips.txt",
    "/root/antizapret/config/forward-ips.txt",
    "/root/antizapret/config/include-adblock-hosts.txt",
    "/root/antizapret/config/exclude-adblock-hosts.txt",
    "/root/antizapret/config/remove-hosts.txt",
]


def extract_client_private_keys(server_conf: str) -> dict[str, str]:
    """
    Parse '# Client = name' + '# PrivateKey = key' pairs from server conf.
    Returns {client_name: private_key_b64}.
    """
    result = {}
    current_name = None
    for line in server_conf.splitlines():
        stripped = line.strip()
        if stripped.startswith("# Client") and "=" in stripped:
            current_name = stripped.split("=", 1)[1].strip()
        elif stripped.startswith("# PrivateKey") and "=" in stripped and current_name:
            priv = stripped.split("=", 1)[1].strip()
            result[current_name] = priv
            current_name = None
    return result


def extract_allowed_ips_from_client_conf(client_conf_dir: str) -> str | None:
    """
    Read any antizapret client conf and extract the AllowedIPs line.
    Returns the full AllowedIPs value string, or None.
    """
    p = Path(client_conf_dir)
    if not p.exists():
        return None
    for f in p.glob("antizapret-*-am.conf"):
        for line in f.read_text().splitlines():
            if line.strip().lower().startswith("allowedips"):
                return line.split("=", 1)[1].strip()
        break  # only need one
    return None


def migrate_files_to_db(db: Session, root: str = "") -> None:
    """
    Migrate files from disk to DB.
    root: prefix for all paths (empty in production, tmpdir in tests).
    """
    store = WgBlobStore(db)
    migrated = 0

    # 1. Migrate managed files
    for rel_path in MANAGED_FILES:
        abs_path = os.path.join(root, rel_path.lstrip("/"))
        if not os.path.exists(abs_path):
            log.warning("File not found, skipping: %s", abs_path)
            continue
        content = Path(abs_path).read_bytes()
        store.put(rel_path, content, by="migrate")
        migrated += 1
        log.info("Migrated %s (%d bytes)", rel_path, len(content))

    # 2. Migrate server keypair from /etc/wireguard/key
    key_path = os.path.join(root, "etc/wireguard/key")
    if os.path.exists(key_path):
        key_content = Path(key_path).read_text()
        priv = pub = None
        for line in key_content.splitlines():
            if line.startswith("PRIVATE_KEY="):
                priv = line.split("=", 1)[1].strip()
            elif line.startswith("PUBLIC_KEY="):
                pub = line.split("=", 1)[1].strip()

        if priv and pub:
            for iface in ("antizapret", "vpn"):
                existing = db.get(WgServerKeys, iface)
                if existing:
                    existing.private_key = priv
                    existing.public_key = pub
                else:
                    db.add(WgServerKeys(iface=iface, private_key=priv, public_key=pub))
            db.commit()
            log.info("Migrated server keypair to wg_server_keys")

    # 3. Extract client private keys and update vpn_configs.config_metadata
    az_path = os.path.join(root, "etc/wireguard/antizapret.conf")
    if os.path.exists(az_path):
        server_conf = Path(az_path).read_text()
        client_keys = extract_client_private_keys(server_conf)
        if client_keys:
            from app.db.models import VPNConfig
            updated = 0
            for config in db.query(VPNConfig).all():
                priv = client_keys.get(config.client_name)
                if priv:
                    meta = dict(config.config_metadata or {})
                    meta["private_key"] = priv
                    config.config_metadata = meta
                    updated += 1
            db.commit()
            log.info("Updated %d vpn_configs with client private keys", updated)

    # 4. Extract AllowedIPs from any existing antizapret client conf
    client_dir = os.path.join(root, "root/antizapret/client/amneziawg/antizapret")
    allowed_ips = extract_allowed_ips_from_client_conf(client_dir)
    if allowed_ips:
        store.put("antizapret:allowed_ips", allowed_ips.encode(), by="migrate")
        log.info("Stored AllowedIPs template (%d chars)", len(allowed_ips))

    log.info("Migration complete: %d files migrated", migrated)


if __name__ == "__main__":
    from app.db.session import SessionLocal
    db = SessionLocal()
    try:
        migrate_files_to_db(db)
    finally:
        db.close()
```
